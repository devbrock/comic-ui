# Combobox

## Description

**Combobox** is a searchable select input; use it when there are many options and users need to filter/type-ahead to find a value.

## Usage

```tsx
import { Combobox } from "@devbrock/comic-ui";

export function Example() {
  const items = [
    { value: "spiderman", label: "Spider-Man" },
    { value: "storm", label: "Storm" },
  ] as const;

  const [value, setValue] = React.useState<string | undefined>();
  return <Combobox items={items} value={value} onValueChange={setValue} />;
}
```
